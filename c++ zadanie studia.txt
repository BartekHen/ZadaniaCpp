int HowMuch(int tablicaPC[],int tablicaUser[]){
    int counter=0;

}

int HowMany(int tablicaPC[],int tablicaUser[]){
    int counter = 0;
    
    for(int i=0;i<4;i++){
        for(int j=0;j<4;j++){
            if(i==j){
                continue;
            }else if(tablicaPC[i]==tablicaUser[j]){
                counter+=1;
            }
        }
    }
    return counter;
